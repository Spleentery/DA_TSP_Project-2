244 std::unordered_map<long,Vertex *> vertexis = graph.getVertexSet();
245 for (auto v: vertexis) {
246 v.second->setVisited(false);
247 v.second->setDist(std::numeric_limits<double>::max());
248 v.second->eraseChildren();
249 }
250
251 Vertex *root = graph.findVertex(0);
252 root->setDist(0);
253 MutablePriorityQueue<Vertex> q;
254 q.insert(root);
255 while (!q.empty()) {
256 auto v = q.extractMin();
257 cout<<"working on:"<<v->getId()<<'\n';
258 v->setVisited(true);
259 if(v->getId()!=0) {
260 v->getPath()->getOrig()->addChildren(v->getId());
261 }
262 for (auto &e: v->getAdj()) {
263 Vertex *w = e->getDest();
264 if (!w->isVisited()) {
265 auto oldDist = w->getDist();
266 if (e->getDistance() < oldDist) {
267 w->setDist(e->getDistance());
268 w->setPath(e);
269 if (oldDist == std::numeric_limits<double>::max()) {
270 q.insert(w);
271 } else {
272 q.decreaseKey(w);
273 }
274 }
275 }
276 }
277 }

279 mst_preorder_path.clear();
280 pathRec(root);

308double CPheadquarters::haversineDistance(double latitude1, double longitude1, double latitude2, double longitude2) {
309 double ang_lat=degToRad(latitude2-latitude1);
310 double ang_lon=degToRad(longitude2-longitude1);
311 double a =sin(ang_lat / 2) * sin(ang_lat / 2) +
312 cos(degToRad (latitude1)) * cos(degToRad (latitude2)) *
313 sin(ang_lon / 2) * sin(ang_lon / 2);
314
315 double c = 2 * atan2(sqrt(a), sqrt(1 - a));
316
317 return EarthRadius * c;
318}

350double CPheadquarters::calculateAngle(double Ax, double Ay, double Bx, double By, double Cx, double Cy) {
351 double ABx = Ax - Bx;
352 double ABy = Ay - By;
353 double BCx = Cx - Bx;
354 double BCy = Cy - By;
355
356 double dotProduct = ABx * BCx + ABy * BCy;
357 double crossProduct = ABx * BCy - ABy * BCx; // Compute the cross product
358
359 double magnitudeAB = std::sqrt(ABx * ABx + ABy * ABy);
360 double magnitudeBC = std::sqrt(BCx * BCx + BCy * BCy);
361
362 double angle = std::acos(dotProduct / (magnitudeAB * magnitudeBC));
363
364 // Adjust the angle based on the sign of the cross product
365 if (crossProduct < 0) {
366 angle = 2 * M_PI - angle;
367 }
368
369 return angle;
370}
