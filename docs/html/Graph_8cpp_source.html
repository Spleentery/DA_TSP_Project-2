280bool Graph::hasArticulationPointUtil(Vertex* pCurrentVertex, int time) {
281 int children = 0;
282 int currentVertexIdInt = std::stoi(pCurrentVertex->getId());
283 visited[currentVertexIdInt] = true;
284 visited[currentVertexIdInt] = true;
285
286 disc[currentVertexIdInt] = low[currentVertexIdInt] = ++time;
287
288 for (auto edge : pCurrentVertex->getAdj()) {
289 Vertex* pAdjacentVertex = edge->getDest();
290 int adjacentVertexIdInt = std::stoi(pAdjacentVertex->getId());
291 if (!visited[adjacentVertexIdInt]) {
292 children++;
293 parent[adjacentVertexIdInt] = currentVertexIdInt;
294
295 if (hasArticulationPointUtil(pAdjacentVertex, time))
296 return true;
297
298 low[currentVertexIdInt] = std::min(low[currentVertexIdInt], low[adjacentVertexIdInt]);
299
300 if (parent[currentVertexIdInt] == -1 && children > 1) {
301 ap[currentVertexIdInt] = true;
302 return true;
303 }
304
305 if (parent[currentVertexIdInt] != -1 && low[adjacentVertexIdInt] >= disc[currentVertexIdInt]) {
306 ap[currentVertexIdInt] = true;
307 return true;
308 }
309 }
310 else if (adjacentVertexIdInt != parent[currentVertexIdInt]) {
311 low[currentVertexIdInt] = std::min(low[currentVertexIdInt], disc[adjacentVertexIdInt]);
312 }
313 }
314
315 return false;
316}
