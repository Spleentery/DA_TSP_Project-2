97void CPheadquarters::heuristicRec(Vertex *v, string route[], unsigned int currentIndex, double distance, unsigned int &nodesVisited, double &totalDistance){
98
99 bool nodesStillUnvisited = false;
100 string id1 = v->getId();
101 Station st1 = stations.find(id1)->second;
102
103 double long1 = st1.get_longitude();
104 double lat1 = st1.get_latitude();
105
106 Vertex *small;
107 double smallAngle = 10000;
108 double x;
109 double y;
110 double angle;
111 double dist;
112
113 for (const auto &edge: v->getAdj()) {
114 Vertex *v2 = edge->getDest();
115 double dist2 = edge->getDistance();
116 if(v2->isVisited() == false){
117 nodesStillUnvisited = true;
118 string id2 = edge->getDest()->getId();
119 Station st2 = stations.find(id2)->second;
120
121 double long2 = st2.get_longitude();
122 double lat2 = st2.get_latitude();
123
124 x = long1 - long2;
125 y = lat1 - lat2;
126
127 angle = atan2(y,x);
128
129 if(angle < smallAngle){
130 smallAngle = angle;
131 small = v2;
132 dist = dist2;
133 }
134
135 }
136 }
137
138 bool inRoute = false;
139 for(int i = 0; i < currentIndex; i++){
140 if(route[i] == v->getId()){
141 inRoute = true;
142 }
143 }
144
145 if(nodesStillUnvisited){
146 route[currentIndex] = small->getId();
147 small->setVisited(true);
148
149 heuristicRec(small, route, currentIndex+1, distance + dist, nodesVisited, totalDistance);
150 }
151 else{
152 nodesVisited = currentIndex;
153 totalDistance = distance;
